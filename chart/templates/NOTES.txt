FastAPI Chassis has been deployed.

1. Get the application URL:
{{- if .Values.ingress.enabled }}
{{- range $host := .Values.ingress.hosts }}
  http{{ if $.Values.ingress.tls }}s{{ end }}://{{ $host.host }}
{{- end }}
{{- else if contains "NodePort" (.Values.service.type | default "ClusterIP") }}
  export NODE_PORT=$(kubectl get -n {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "fastapi-chassis.fullname" . }})
  export NODE_IP=$(kubectl get nodes -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT
{{- else if contains "LoadBalancer" (.Values.service.type | default "ClusterIP") }}
  export SERVICE_IP=$(kubectl get svc -n {{ .Release.Namespace }} {{ include "fastapi-chassis.fullname" . }} --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo http://$SERVICE_IP:{{ .Values.service.port | default 80 }}
{{- else }}
  kubectl port-forward -n {{ .Release.Namespace }} svc/{{ include "fastapi-chassis.fullname" . }} 8000:{{ .Values.service.port | default 80 }}
  echo http://localhost:8000
{{- end }}

2. Check health:
  curl http://localhost:8000{{ .Values.app.healthCheckPath | default "/healthcheck" }}

3. Check readiness:
  curl http://localhost:8000{{ .Values.app.readinessCheckPath | default "/ready" }}
